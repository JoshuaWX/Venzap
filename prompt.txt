═══════════════════════════════════════════════════════════
                    PROJECT OVERVIEW
═══════════════════════════════════════════════════════════

Venzap is a two-sided chat-commerce platform that connects
local vendors of ANY category to customers through chat —
starting with Telegram and expanding to WhatsApp and other
platforms as funding allows.

SIDE 1 — VENDORS (Web Dashboard)
  Any local vendor — food, grocery, pharmacy, laundry,
  fashion, or any catalogue-based business — registers
  on a web dashboard, manages their catalogue/menu and
  prices, and receives incoming orders in real time.

  LAUNCH PHASE → Food vendors only.
  Food has the highest order frequency and fastest proof
  of concept. But the entire platform architecture is
  built from day one to support ALL vendor categories.
  Expansion to grocery, pharmacy, fashion, laundry etc.
  begins immediately after launch traction is proven.

SIDE 2 — USERS (Telegram Bot — Phase 1)
  Users sign up entirely inside Telegram, receive a
  dedicated Nigerian virtual bank account (powered by
  Payaza Dynamic Virtual Account API) the moment their
  account is verified, fund their Venzap wallet by doing
  a plain bank transfer from any Nigerian bank app, browse
  registered vendors, and place orders — all without ever
  downloading a new app.

  WHY TELEGRAM NOW:
  Telegram provides a completely free, unrestricted bot API
  with zero business verification requirements and zero
  per-message fees. This allows us to build, deploy and
  iterate at maximum speed during the hackathon and early
  growth phase.

  EXPANSION PLAN (with funding):
  → Phase 2: WhatsApp Business API (largest Nigerian user base)
  → Phase 3: Any other widely adopted messaging platform
  → The backend is designed to plug in new chat channels
    without rebuilding core logic.

WALLET INFRASTRUCTURE → Payaza Dynamic Virtual Account API
  Every Venzap user gets a real, unique Nigerian bank account
  number (e.g. Wema Bank 0123456789) the moment they verify
  their email. They fund their wallet by simply transferring
  money to that account from GTB, Access, Opay, Kuda or any
  bank app — exactly like sending money to a friend.
  Payaza detects the transfer, fires a webhook, and the
  wallet is credited instantly. No payment links to click.
  No extra steps. This is the core financial UX innovation.

PAYMENT INFRASTRUCTURE → Payaza API (full suite)
  - Dynamic Virtual Account API → wallet funding
  - Payment Link API → fallback funding method
  - Webhook API → real-time payment notifications
  - Transfer API → vendor payouts (future)
  All money on the platform flows through Payaza.

AI INFRASTRUCTURE → Hybrid LLM + Rule-Based System
  Layer 1: GPT-4o mini interprets all natural language input
  Layer 2: Deterministic rule engine executes ALL actions
  The LLM NEVER touches money, payments or order execution.
  It ONLY parses intent and extracts structured data.
  If LLM fails for ANY reason → rule engine takes over silently.
  Users never experience any degradation in service.

async def handle_payaza_webhook(request: Request):
    raw_body = await request.body()
    signature = request.headers.get("X-Payaza-Signature", "")

    # Verify signature FIRST — before reading payload
    if not verify_hmac_sha512(raw_body, signature, PAYAZA_SECRET_KEY):
        logger.warning("Webhook signature verification failed")
        return {"status": "ok"}  # Return 200 always

    payload = json.loads(raw_body)
    payaza_ref = payload.get("reference") or payload.get("id")

    # Return 200 IMMEDIATELY — dispatch async
    process_payaza_webhook.delay(payload, payaza_ref)
    return {"status": "ok"}


@celery_app.task(bind=True, max_retries=3)
def process_payaza_webhook(self, payload: dict, payaza_ref: str):
    # Idempotency — never double-process
    existing = db.get_webhook_event(payaza_ref)
    if existing and existing.processed:
        return

    db.create_webhook_event(payaza_ref, payload)

    event_type = payload.get("event") or payload.get("type")
    try:
        if event_type == "virtual_account.credit":
            virtual_account_service.handle_dva_credit(payload)
        elif event_type == "payment.success":
            wallet_service.handle_payment_link_credit(payload)
        elif event_type == "transfer.success":
            vendor_payout_service.handle_transfer_success(payload)
        else:
            logger.info(f"Unhandled webhook: {event_type}")
        db.mark_webhook_processed(payaza_ref)
    except Exception as exc:
        db.log_webhook_error(payaza_ref, str(exc))
        raise self.retry(exc=exc, countdown=60)

def verify_hmac_sha512(raw_body: bytes, signature: str, secret: str) -> bool:
    expected = hmac.new(
        secret.encode("utf-8"),
        raw_body,
        hashlib.sha512
    ).hexdigest()
    return hmac.compare_digest(expected, signature)

def sanitize_input(text: str) -> str | None:
    """Returns clean text or None (→ fallback immediately)."""

    if len(text) > 500:
        return None

    text = re.sub(r'<[^>]+>', '', text)

    INJECTION_PATTERNS = [
        r"ignore\s+(all\s+)?(your\s+)?(previous\s+)?(the\s+)?instructions",
        r"you\s+are\s+now",
        r"new\s+instruction",
        r"system\s*prompt",
        r"forget\s+(everything|all|your\s+rules)",
        r"pretend\s+(you\s+are|to\s+be)",
        r"act\s+as\s+(if|a|an)",
        r"\bDAN\b|\bjailbreak\b|\bdeveloper\s+mode\b",
        r"disregard|override\s+rules|bypass\s+security",
        r"reveal\s+your\s+(prompt|instructions|system)",
        r"what\s+are\s+your\s+(instructions|rules|prompt)",
    ]
    for pattern in INJECTION_PATTERNS:
        if re.search(pattern, text, re.IGNORECASE):
            return None

    text = " ".join(text.split()).strip()
    return text if text else None

class GuardrailValidator:

    ALLOWED_INTENTS = {
        "greet","list_vendors","select_vendor","view_catalogue",
        "select_items","remove_item","view_cart","clear_cart",
        "confirm_order","cancel_order","enter_address",
        "check_balance","fund_wallet","view_account",
        "order_status","order_history","faq","unclear","out_of_scope"
    }

    def validate(self, llm_output, active_vendors, vendor_catalogue):

        # CHECK 1: Valid JSON
        try:
            data = json.loads(llm_output)
        except Exception:
            return GuardrailResult.fail("JSON_PARSE_ERROR")

        # CHECK 2: Schema valid + correct types
        required = {
            "intent": str, "vendor_name": (str, type(None)),
            "items": list, "address": (str, type(None)),
            "faq_answer": (str, type(None)),
            "clarification": (str, type(None)), "confidence": float,
        }
        for field, t in required.items():
            if field not in data:
                return GuardrailResult.fail(f"MISSING_FIELD:{field}")
            if not isinstance(data[field], t):
                return GuardrailResult.fail(f"WRONG_TYPE:{field}")

        # CHECK 3: Intent in allowed enum
        if data["intent"] not in self.ALLOWED_INTENTS:
            return GuardrailResult.fail("INVALID_INTENT")

        # CHECK 4: Vendor exists in active list
        if data["vendor_name"] is not None:
            valid_names = [v["name"] for v in active_vendors]
            if data["vendor_name"] not in valid_names:
                data["vendor_name"] = None
                data["intent"] = "unclear"
                data["clarification"] = "Which vendor would you like?"

        # CHECK 5: Items exist in vendor catalogue
        if data["items"] and vendor_catalogue:
            valid = {i["name"] for i in vendor_catalogue}
            data["items"] = [i for i in data["items"]
                             if i.get("name") in valid]
            if not data["items"]:
                data["intent"] = "unclear"
                data["clarification"] = (
                    "I couldn't find those items. What would you like?"
                )

        # CHECK 6: Quantities sane (1-20)
        for item in data.get("items", []):
            qty = item.get("quantity", 1)
            if not isinstance(qty, int) or not (1 <= qty <= 20):
                item["quantity"] = 1

        # CHECK 7: Confidence threshold
        if data["confidence"] < CONFIDENCE_THRESHOLD:
            data["intent"] = "unclear"
            data["clarification"] = (
                data.get("clarification") or
                "Could you rephrase that? I want to get it right!"
            )

        # CHECK 8: Scan text fields for injected content
        FORBIDDEN = [
            r"system\s*prompt", r"ignore\s+instructions",
            r"wallet\s*balance", r"₦\d+",
            r"\d{10}", r"account\s*number",
            r"password", r"api[_\s]?key",
            r"<script", r"javascript:",
        ]
        for field in ["faq_answer", "clarification"]:
            val = data.get(field) or ""
            for pat in FORBIDDEN:
                if re.search(pat, val, re.IGNORECASE):
                    return GuardrailResult.fail(f"INJECTION_IN:{field}")

        return GuardrailResult.pass_(data)

# These states ALWAYS go to rule engine — LLM never sees them
RULE_ENGINE_ONLY_STATES = [
    AWAIT_PASSWORD, AWAIT_OTP, CONFIRM_ORDER,
    AWAIT_AMOUNT, PROCESS_PAYMENT, AWAIT_NEW_PASSWORD,
]

async def route_message(update, context):
    user_id = update.effective_user.id
    text = update.message.text
    current_state = await redis_state.get_state(user_id)

    # Button taps → rule engine always
    if update.callback_query:
        return await rule_engine.handle_callback(update, context)

    # Sensitive states → rule engine always
    if current_state in RULE_ENGINE_ONLY_STATES:
        return await rule_engine.handle(update, context)

    # Free text → sanitize → LLM → guardrails → rule engine
    sanitized = sanitize_input(text)
    if sanitized is None:
        await log_ai(user_id, text, sanitized=False, fallback=True)
        return await fallback_handler(update, context)

    try:
        intent = await asyncio.wait_for(
            ai_service.parse_intent(sanitized, user_id, current_state),
            timeout=5.0
        )
        if intent.is_valid:
            return await rule_engine.handle_intent(intent, update, context)
        await log_ai(user_id, sanitized, fallback=True,
                     error=intent.failure_reason)
        return await fallback_handler(update, context)

    except asyncio.TimeoutError:
        await log_ai(user_id, sanitized, fallback=True, error="LLM_TIMEOUT")
        return await fallback_handler(update, context)
    except Exception as e:
        await log_ai(user_id, sanitized, fallback=True, error=str(e))
        return await fallback_handler(update, context)
